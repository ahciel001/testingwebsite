function generateMarkdown() {

  const desc = document.getElementById("description").value;
  const start = document.getElementById("startTime").value;
  const dur = document.getElementById("duration").value;
  const difficulty = document.getElementById("difficulty").value;
  const permaDeath = document.getElementById("permaDeath").textContent;

  const playerCount = parseInt(document.getElementById("playerCount").value) || 0;

  const existingNames = [...document.querySelectorAll(".playerEntry")]
    .map(p => "- " + (p.value || p.placeholder));

  const apl = [...document.querySelectorAll(".apl:checked")].map(c => c.value).join(" ");

**Start time:** ${start}

**Expected duration:** ${dur} hours  
**Expected APL:** ${apl}  
**Difficulty:** ${difficulty}, Perma Death: ${permaDeath}  
**Session Focus:** ${focus}

**House rules:**  
${houseRules}

**Player count:** ${playerCount}

**Players:**  
${existingNames.join("\n")}`;

  document.getElementById("output").value = markdown;
}
